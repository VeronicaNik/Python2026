def delivery_price(city: str, weight_kg: float, urgent: bool) -> float:
    if urgent:
        return 500 + 30 * weight_kg
    else:
        return 300 + 20 * weight_kg

# Позиционные вызовы
delivery_price('moskva', 10, True)
delivery_price("piter", 5, False)

# Именованные вызовы
delivery_price(city="Ova", weight_kg=7, urgent=True)
delivery_price(city="Nova", weight_kg=3, urgent=False)
