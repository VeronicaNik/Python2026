secret_number = 15
attempt = None

while attempt != secret_number:
    try:
        attempt = int(input())
        if attempt == 0:
            print("Выход по желанию пользователя")
            break
        if attempt < secret_number:
            print("Мало")
        elif attempt > secret_number:
            print("Много")
        else:
            print("Угадано")
    except ValueError:
        continue  # Или ничего не выводить, как в примере задания
